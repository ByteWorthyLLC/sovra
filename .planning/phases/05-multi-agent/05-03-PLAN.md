  <behavior>
    Conversation memory:
    - buildConversationMemory fetches all messages for a conversation from DB
    - Returns them as CoreMessage[] in chronological order
    - Prepends system prompt if agent has one

    Summary memory:
    - buildSummaryMemory fetches messages, takes oldest N, calls generateText to summarize
    - Returns [system summary message, ...recent messages]
    - Summary prompt: "Summarize this conversation in under 200 words, preserving key decisions and facts."

    Vector memory:
    - buildVectorMemory takes current prompt, generates embedding (mock in tests), calls hybridSearch
    - Returns matching results as system context messages + recent messages

    Hybrid memory:
    - buildHybridMemory calls buildSummaryMemory for compressed history + buildVectorMemory for semantic retrieval
    - Merges both into a single CoreMessage[] array (summary first, then vector results, then recent)

    Context compression:
    - estimateTokens(text) returns rough token count at 4 chars per token
    - snipMessages(messages, maxMessages) keeps system messages + last N conversational messages
    - microcompactMessages(model, messages, keepLast) summarizes older messages into a single system message
    - collapseMessages(messages, keepLast) keeps only the last N messages, discards all else
  </behavior>

       ```typescript
       import type { SupabaseClient } from '@supabase/supabase-js'
       import type { LanguageModel } from 'ai'

       export type CoreMessage = { role: 'system' | 'user' | 'assistant'; content: string }

       export interface MemoryBuildParams {
         supabase: SupabaseClient
         conversationId: string
         tenantId: string
         workspaceId?: string
         strategy: 'conversation' | 'summary' | 'vector' | 'hybrid'
         maxTokenBudget: number
         currentPrompt: string
         systemPrompt?: string
         model?: LanguageModel
       }

       export async function buildContextMessages(params: MemoryBuildParams): Promise<CoreMessage[]> {
         switch (params.strategy) {
           case 'conversation': return (await import('./conversation')).buildConversationMemory(params)
           case 'summary': return (await import('./summary')).buildSummaryMemory(params)
           case 'vector': return (await import('./vector')).buildVectorMemory(params)
           case 'hybrid': return (await import('./hybrid')).buildHybridMemory(params)
         }
       }
       ```
